## 🤖 4. Script para Integración Futura (Pipeline de Automatización)

Si en el futuro deseas exportar esta lista de manera automática a una tabla de **Excel (`.xlsx`)**, a un archivo **CSV** para integrarlo con tu ERP, o a una base de datos, te dejamos este bloque de código escrito en Python. 

```python
import json
import struct
import csv

def extraer_bom_desde_glb(ruta_glb, ruta_csv_salida):
    with open(ruta_glb, 'rb') as f:
        # Saltar cabecera GLB (12 bytes)
        f.seek(12)
        # Leer longitud de la cabecera JSON (Chunk 0)
        chunk_length = struct.unpack('<I', f.read(4))[0]
        f.seek(20) # Saltar tipo de chunk
        # Leer JSON
        gltf_data = json.loads(f.read(chunk_length).decode('utf-8'))
        
    # Agrupar y limpiar piezas
    piezas = []
    for node in gltf_data.get('nodes', []):
        name = node.get('name')
        if name:
            clean_name = name.split(':')[:-1][0].strip() if ':' in name else name.strip()
            piezas.append(clean_name)
            
    # Contar ocurrencias (Cantidades)
    from collections import Counter
    conteo_piezas = Counter(piezas)
    
    # Exportar a CSV limpio
    with open(ruta_csv_salida, mode='w', newline='', encoding='utf-8') as archivo_csv:
        escritor = csv.writer(archivo_csv)
        escritor.writerow(["Codigo_Referencia", "Cantidad_Requerida"])
        for ref, cant in conteo_piezas.items():
            escritor.writerow([ref, cant])
            
    print(f"¡BOM automatizada exportada exitosamente a: {ruta_csv_salida}!")
